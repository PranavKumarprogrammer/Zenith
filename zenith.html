Developer Guide

Complete setup and integration guide

Installation & Setup

1

Clone Repository

  • Download the project files
  • Node.js 16+ required
  • npm or yarn package manager
2

Install Dependencies

npm install

Installs: express, cors, uuid, jsonwebtoken

3

Start Server

npm start

Runs on http://localhost:3000

4

Initialize SDK

const ZMClient = require('./sdk');
const client = new ZMClient();
await client.register('email', 'pass');
5

Create Bucket

client.jwt = token;
const bucket = await client
  .createBucket('my-data');
6

Store & Retrieve Data

await client.put(bucketId, '/path', data);
const data = await client.get(bucketId, '/path');

Core API Methods

Register User 
await client.register(
  'dev@example.com',
  'password123',
  'Developer'
);
// Returns: { userId, token }
Create Bucket 
const bucket = await client.createBucket(
  'my-data',
  {
    durability: 'absolute',
    region: 'us-east-1'
  }
);
// Returns: { bucketId, name }
Write Data (PUT) 
await client.put(
  bucketId,
  '/users/john',
  {
    name: 'John Doe',
    email: 'john@example.com'
  }
);
// Data encrypted & stored
Read Data (GET) 
const data = await client.get(
  bucketId,
  '/users/john'
);
console.log(data);
// { name, email, ... }
List Items 
const items = await client
  .listItems(bucketId);
// Returns array of items with metadata
// { path, size, createdAt }
Batch Write 
await client.batchWrite(bucketId, [
  { path: '/user/1', data: {...} },
  { path: '/user/2', data: {...} },
  { path: '/user/3', data: {...} }
]);
// All-or-nothing transaction
Vector Search 
const results = await client
  .vectorQuery(bucketId, {
    query: 'find similar',
    topK: 10
  });
// Semantic search results
Delete Data 
await client.delete(
  bucketId,
  '/users/john'
);
// Securely removes data
Get Statistics 
const stats = await client.getStats();
// {
//   bucketsCount: 5,
//   totalItems: 1000,
//   latency: 2.3ms,
//   uptime: '99.99%'
// }

const ZMClient = require('./sdk');

async function demo() {
  const client = new ZMClient();
  const user = await client.register(
    'dev@example.com',
    'pass123'
  );
  client.jwt = user.token;
  const bucket = await client
    .createBucket('data');
  await client.put(bucket.bucketId,
    '/test', {test: 'value'});
}

demo();

Bucket Operations

POST /api/buckets

Create new bucket for data storage

Headers:
Authorization: Bearer TOKEN

Body:
{
  "name": "my-bucket",
  "durability": "absolute",
  "region": "us-east-1"
}

GET /api/buckets

List all your buckets

Headers:
Authorization: Bearer TOKEN

Response: {
  "buckets": [
    {
      "bucketId": "uuid",
      "name": "my-bucket",
      "itemCount": 5
    }
  ]
}

Data Operations

PUT /api/buckets/:id/data/*

Write or update data

PUT /buckets/abc123/data/users/john

Body: Any JSON data
{ "name": "John", "email": "..." }

Response: {
  "success": true,
  "path": "/users/john"
}

GET /api/buckets/:id/data/*

Read data by path

GET /buckets/abc123/data/users/john

Response: {
  "data": { ... },
  "metadata": {
    "createdAt": "timestamp",
    "updatedAt": "timestamp",
    "size": 256
  }
}

DELETE /api/buckets/:id/data/*

Delete data by path

DELETE /buckets/abc123/data/users/john

Response: {
  "success": true,
  "message": "Data deleted successfully"
}

GET /api/buckets/:id/items

List all items in bucket

Response: {
  "bucketId": "abc123",
  "itemCount": 5,
  "items": [
    {
      "path": "/users/john",
      "size": 256,
      "createdAt": "timestamp"
    }
  ]
}
